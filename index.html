Step 1: Create or Open Your Google Sheet

Go to sheets.google.com and create a new sheet or open an existing one.

Step 2: Open Apps Script

In your Google Sheet, click:

Extensions ‚Üí Apps Script

Step 3: Paste This Code

Delete any existing code and paste this:

function doPost(e) {
  const ss = SpreadsheetApp.getActiveSpreadsheet();
  let sheet = ss.getSheetByName('Commander Game Log');
  
  // Create the sheet if it doesn't exist
  if (!sheet) {
    sheet = ss.insertSheet('Commander Game Log');
    // Add headers
    sheet.appendRow([
      'Date',
      'Bracket',
      'Your Commander',
      'Opponent 1',
      'Opponent 2',
      'Opponent 3',
      'Went First',
      'Winner',
      'Turn Ended'
    ]);
    // Format headers
    const headerRange = sheet.getRange(1, 1, 1, 9);
    headerRange.setFontWeight('bold');
    headerRange.setBackground('#667eea');
    headerRange.setFontColor('#ffffff');
  }
  
  const data = JSON.parse(e.postData.contents);
  
  sheet.appendRow([
    data.date,
    data.bracket,
    data.myCommander,
    data.opponent1,
    data.opponent2,
    data.opponent3,
    data.wentFirst,
    data.winner,
    data.turnEnded
  ]);
  
  return ContentService
    .createTextOutput(JSON.stringify({status: 'success'}))
    .setMimeType(ContentService.MimeType.JSON);
}
